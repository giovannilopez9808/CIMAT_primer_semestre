\subsection*{Problema 7}
\textbf{Sea $X$ una variable aleatoria con la siguiente distribución
    \begin{table}[H]
        \centering
        \begin{tabular}{l|cccc}
                     & 1   & 2   & 3   & 4   \\ \hline
            $P(X=x)$ & 0.2 & 0.1 & 0.4 & 0.3
        \end{tabular}
    \end{table}
    y $Y\sim X$ e independiente de $X$. Calcula $P(X>2|Y<3)$ y $P(X\neq Y)$.
}

\begin{itemize}
    \item \textbf{Calcula} $P(X>2|Y<3)$

          Al ser $X,Y$ variables aleatorias independientes entonces:
          \begin{align*}
              P(X>2|Y<3) & = P(X>2)         \\
                         & = P(X=3) +P(X=4) \\
                         & = 0.4+0.3        \\
                         & = 0.7
          \end{align*}

          Por lo tanto:
          \begin{equation*}
              P(X>2|Y<3)=0.7
          \end{equation*}

          Realizando el calculo de $P(X\neq Y)$.

          El cálculo de este problema es complejo, ya que para cada $X$ seleccionada correspondera a tres opciones de $Y$, cada una con su probabilidad. Es por ello que se aplicará lo siguiente:
          \begin{equation*}
              P(X\neq Y) = 1-P(X=Y)
          \end{equation*}

    \item \textbf{Calcula} $P(X\neq Y)$

          Realizando este cálculo se obtiene lo siguiente:
          \begin{equation}
              P(X=Y) = \sum_i P(X=x_i,Y=x_i)
              \label{eq:pxy}
          \end{equation}
          como $X,Y$ son variables aleatorias independientes, entonces
          \begin{equation*}
              P(X=x_i,Y=x_i) = P(X=x_i)P(Y=x_i)
          \end{equation*}
          entonces, la ecuación \ref{eq:pxy} se puede escrbir como
          \begin{align*}
              P(X=Y) & = \sum_i P(X=x_i,Y=x_i)                      \\
                     & =\sum P(X=x_i,Y=x_i)                         \\
                     & =P(X=1,Y=1)+P(X=2,Y=2)+P(X=3,Y=3)+P(X=4,Y=4) \\
                     & = 0.2*0.2+0.1*0.1+0.4*0.4+0.3*0.3            \\
                     & = 0.3
          \end{align*}
          Por lo tanto:
          \begin{align*}
              P(X\neq Y) & = 1- P(X=Y) \\
                         & = 1-0.3     \\
              P(X\neq Y) & = 0.7
          \end{align*}
\end{itemize}
